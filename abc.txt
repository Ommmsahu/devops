name = input("Enter student name: ")
marks = []

for i in range(5):
    m = int(input(f"Enter mark {i+1}: "))
    marks.append(m)

total = sum(marks)
average = total / len(marks)

if average >= 90:
    grade = "A"
elif average >= 75:
    grade = "B"
elif average >= 60:
    grade = "C"
else:
    grade = "D"

print("\nStudent:", name)
print("Total:", total)
print("Average:", average)
print("Grade:", grade)
